### T1 LC7 整数翻转

```python
class Solution:
    def reverse(self, x: int) -> int:
        y,res=abs(x),0
        offset=2**31-1 if x>0 else 2**31
        while y>0:
            res=res*10+y%10
            if res>offset:return 0
            y//=10
        return res if x>0 else -res
```

### T2 LC98 验证二叉搜索树

常规操作，二叉搜索树就是要用他的性质来做

```python
class Solution:
    def isValidBST(self, root: TreeNode) -> bool:
        result=[]

        def preorder(root):
            if not root:return 
            if root.left:preorder(root.left)
            result.append(root.val)
            if root.right:preorder(root.right)
        
        preorder(root)
        for i in range(len(result)-1):
            if result[i]>=result[i+1]:return False
        return True
```

### T3 LC572 另一个树的子树

```python
class Solution:
    def isSubtree(self, s: TreeNode, t: TreeNode) -> bool:
        def subTree(s,t):
            if not s and not t:return True
            if not s or not t:return False
            if s.val!=t.val:return False
            return subTree(s.left,t.left) and subTree(s.right,t.right)
        if not s:return False
        return subTree(s,t) or self.isSubtree(s.left,t) or self.isSubtree(s.right,t)
```
