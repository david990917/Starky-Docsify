### T1 LC983 最低票价

动态规划，倒序生成，需要使用`@lru_cache(None)`

```python
class Solution:
    def mincostTickets(self, days: List[int], costs: List[int]) -> int:
        duration =[1,7,30]
        
        @lru_cache(None)
        def dp(i):
            if i>365:return 0
            elif i in days:
                return min(dp(i+d)+c for c,d in zip(costs,duration))
            else:return dp(i+1)
        return dp(1)
```

### T2 LC1290 二进制链表转整数

```cpp
class Solution {
public:
    int getDecimalValue(ListNode* head) {
        long long result=0;
        while (head) {
            result = result * 2 + head->val;
            head = head->next;
        }
        return result;
    }
};
```

### T3 LC468 验证IP地址

使用基础函数库加正则表达式

```python
import re
class Solution:
    chunk_IPv4 = r'([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])'
    patten_IPv4 = re.compile(r'^(' + chunk_IPv4 + r'\.){3}' + chunk_IPv4 + r'$')
    
    chunk_IPv6 = r'([0-9a-fA-F]{1,4})'
    patten_IPv6 = re.compile(r'^(' + chunk_IPv6 + r'\:){7}' + chunk_IPv6 + r'$')

    def validIPAddress(self, IP: str) -> str:        
        if '.' in IP:
            return "IPv4" if self.patten_IPv4.match(IP) else "Neither" 
        if ':' in IP:
            return "IPv6" if self.patten_IPv6.match(IP) else "Neither" 
        return "Neither"
```
